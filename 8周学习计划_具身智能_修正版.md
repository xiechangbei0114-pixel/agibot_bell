```python
#!/usr/bin/env python3
import rclpy
from rclpy.node import Node

class PandaController(Node):
    def __init__(self):
        super().__init__('panda_controller')
        self.get_logger().info('Panda 控制器已启动！')

def main():
    rclpy.init()
    node = PandaController()
    rclpy.spin_once(node, timeout_sec=1)
    print("✅ ROS2 + MoveIt2 环境正常！")
    print("当前阶段：请在 RViz 中拖动机械臂末端，观察关节变化。")
    node.destroy_node()
    rclpy.shutdown()

if __name__ == '__main__':
    main()
```

> ⚠️ `colcon build` 报错缺依赖时，先 `rosdep install --from-paths src --ignore-src -r -y`。网络慢则挂代理或换清华源。
